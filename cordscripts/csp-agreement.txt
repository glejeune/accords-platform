	function BusinessValue( $v )
	{
		$bv = $v.get();
		$m = "&lt;business";
		$sn = $bv["occi.business.name"];
		if ( $sn.length() > 0 )
		{
			$m = $m#' name="'#$sn#'"';
		}
		$sn = $bv["occi.business.nature"];
		if ( $sn.length() > 0 )
		{
			$m = $m#' nature="'#$sn#'"';
		}
		$sn = $bv["occi.business.type"];
		if ( $sn.length() > 0 )
		{
			$m = $m#' type="'#$sn#'"';
		}
		$sn = $bv["occi.business.expression"];
		if ( $sn.length() > 0 )
		{
			$m = $m#' expression="'#$sn#'"';
		}
		$m = $m#"/&gt;";
		$m.display();
	}

	function ServiceGuarantee( $g )
	{
		$sg = $g.get();
		$m = "&lt;guarantee";
		$sn = $sg["occi.guarantee.name"];
		if ( $sn.length() > 0 )
		{
			$sm = $sm#' name="'#$sn#'"';
		}
		$sn = $sg["occi.guarantee.scope"];
		if ( $sn.length() > 0 )
		{
			$m = $m#' scope="'#$sn#'"';
		}
		$sn = $sg["occi.guarantee.obligated"];
		if ( $sn.length() > 0 )
		{
			$m = $m#' obligated="'#$sn#'"';
		}
		$sn = $sg["occi.guarantee.importance"];
		if ( $sn.length() > 0 )
		{
			$m = $m#' importance="'#$sn#'"';
		}
		$m = $m#"&gt;";
		$m.display();
		$sn = $sg["occi.guarantee.variable"];
		if ( $sn.length() > 0 )
		{
			ServiceVariable($sn);
		}
		foreach ( $sg as $bvi )
		{
			forboth ( $bvi as $bvn and $bvv )
			{
				if ( $bvn =="link" )
				{
					BusinessValue( $bvv );
				}
			}
		}
		"&lt;/guarantee&gt;".display();
	}

	function ServiceTerms( $t )
	{

		foreach ( $t as $tt )
		{
			forboth ( $tt as $ttn and $ttv )
			{
				if ( $ttn == "link" )
				{
					$st = $ttv.get();
					$sm = "&lt;term";
					$sn = $st["occi.term.name"];
					if ( $sn.length() > 0 )
					{
						$sm = $sm#' name="'#$sn#'"';
					}
					$sm = $sm#"&gt;";
					$sm.display();
					$smi = $st["occi.term.manifest"];
					if ( $smi.length() > 0 )
					{
						$smm = $smi.get();
						$sm = "&lt;manifest";
						$smn = $smm["occi.manifest.name"];
						if ( $smn.length() > 0 )
						{
							$sm = $sm#' name="'#$smn#'"';
						}
						$sm = $sm#"/&gt;";
						$sm.display();
					}
					$spi = $st["occi.term.provider"];
					if ( $spi.length() > 0 )
					{
						$spm = $spi.get();
						$sm = "&lt;provider";
						$spn = $spm["occi.provider.name"];
						if ( $spn.length() > 0 )
						{
							$sm = $sm#' name="'#$spn#'"';
						}
						$sm = $sm#"/&gt;";
						$sm.display();
					}
					"&lt;/term&gt;".display();
				}
			}
		 }
	}

					$tm = "&lt;terms";

					if ( $tn.length() > 0 )

						$tm = $tm#' name="'#$tn#'"';
					}
					$tn = $terms["occi.terms.type"];
					if ( $tn.length() > 0 )

						$tm = $tm#' type="'#$tn#'"';
